Kode Roblox: Sell Platform

-- Menyimpan Part Platform ke dalam Variable local sellPart = script.Parent local function sellItems(playerEmas, playerUang) -- Mengalikan Item yang didapat dengan 100 gold local totalSell = playerEmas.Value * 100 -- Menambahkan Nilai Gold playerUang.Value = playerUang.Value + totalSell -- Reset Nilai Item playerEmas.Value = 0 end local function onTouch(partTouched) local character = partTouched.Parent local humanoid = character:FindFirstChildWhichIsA("Humanoid") if humanoid then local player = game.Players:GetPlayerFromCharacter(humanoid.Parent) -- Mendapatkan Leaderboard Player local playerStats = player:FindFirstChild("leaderstats") -- Mendapatkan Gold dan Item local playerEmas = playerStats:FindFirstChild("Emas") local playerUang = playerStats:FindFirstChild("Uang") -- Cek apakah Player menyentuh SellPlatform print("A player touched sell platform") sellItems(playerEmas, playerUang) end end sellPart.Touched:Connect(onTouch)
